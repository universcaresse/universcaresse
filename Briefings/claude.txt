javascript// ========================================
// SAUVEGARDE AUTOMATIQUE

function sauvegarderSheet() {
  try {
    const ss        = SpreadsheetApp.openById('16Syw5XypiHauOMpuAu-bWfIMMnMObn9avqoSEYjaNu0');
    const nomFichier = 'Univers Caresse — Sauvegarde ' + Utilities.formatDate(new Date(), 'America/Toronto', 'yyyy-MM-dd');
    const nomDossier = 'Univers Caresse — Sauvegardes';

    // Trouver ou créer le dossier
    let dossier;
    const dossiers = DriveApp.getFoldersByName(nomDossier);
    if (dossiers.hasNext()) {
      dossier = dossiers.next();
    } else {
      dossier = DriveApp.createFolder(nomDossier);
    }

    // Copier le fichier
    const fichier = DriveApp.getFileById(ss.getId());
    fichier.makeCopy(nomFichier, dossier);

    // Supprimer les copies au-delà de 28
    const copies = [];
    const fichiers = dossier.getFiles();
    while (fichiers.hasNext()) {
      const f = fichiers.next();
      copies.push({ fichier: f, date: f.getDateCreated() });
    }
    copies.sort((a, b) => a.date - b.date);
    while (copies.length > 28) {
      copies.shift().fichier.setTrashed(true);
    }

    // Courriel hebdomadaire — chaque dimanche
    const jour = new Date().getDay(); // 0 = dimanche
    if (jour === 0) {
      MailApp.sendEmail({
        to:      'universcaresse@outlook.com',
        subject: '[Univers Caresse] Sauvegarde hebdomadaire — ' + Utilities.formatDate(new Date(), 'America/Toronto', 'yyyy-MM-dd'),
        body:    'Bonjour,\n\nLa sauvegarde hebdomadaire a été effectuée avec succès.\n\nFichier : ' + nomFichier + '\nNombre de copies conservées : ' + Math.min(copies.length, 28) + '/28\nDossier : ' + nomDossier + ' (Google Drive)\n\nUnivers Caresse — Sauvegarde automatique'
      });
    }

    return { success: true, fichier: nomFichier };
  } catch(e) {
    return { success: false, message: e.message };
  }
}

function creerDeclencheurSauvegarde() {
  // Supprimer les déclencheurs existants pour éviter les doublons
  ScriptApp.getProjectTriggers().forEach(t => {
    if (t.getHandlerFunction() === 'sauvegarderSheet') ScriptApp.deleteTrigger(t);
  });
  // Créer un déclencheur quotidien à 2h du matin
  ScriptApp.newTrigger('sauvegarderSheet')
    .timeBased()
    .everyDays(1)
    .atHour(2)
    .create();
}
